def decode(el):
    if el is None: return ''
    if el.get('base64') == 'true': return base64.b64decode(el.text).decode('utf-8', errors='replace')
    return el.text or ''
def body(s):
    i = s.find('\r\n\r\n')
    return s[i+4:] if i != -1 else s
def cookie(s):
    for line in s.split('\r\n'):
        if line.lower().startswith('cookie:'): return line[8:].strip()
    return ''
